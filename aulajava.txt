    private void btcalcularActionPerformed(java.awt.event.ActionEvent evt) {                                           
        SalarioColaborador william = new SalarioColaborador();
        float sal = slsalario.getValue();
        william.setAumento(sal);
        
       float salario = william.getSalario();
       int salarioa = Math.round(salario);
       String salariob = Integer.toString(salarioa);
        lblsalario.setText(salariob);
        
        float aumento = william.getAumento();
       int aumentoa = Math.round(aumento);
       String aumentob = Integer.toString(aumentoa);
        lblaumento.setText(aumentob);
       
        float salarioaumento = william.getValorAumento();
       int salarioaumentoa = Math.round(salarioaumento);
       String salarioaumentob = Integer.toString(salarioaumentoa);
        lblsalarioaumento.setText(salarioaumentob);
       
        String percentual = william.getPercentual();
        lblpercentual.setText(percentual);
        
        float salariocorrigido = william.getValorReal();
       int salariocorrigidoa = Math.round(salariocorrigido);
       String salariocorrigidob = Integer.toString(salariocorrigidoa);
        lblinflacao.setText(salariocorrigidob);
        
    }
